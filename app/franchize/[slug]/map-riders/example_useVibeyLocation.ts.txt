"use client";

/**
 * useVibeyLocation - The Ultimate Telegram Location Hook
 * 
 * Combines multiple location sources for maximum reliability:
 * 1. Mini App Geolocation API (primary, most accurate)
 * 2. Telegram Bot Live Location (fallback, works when Mini App closed)
 * 3. Manual location sharing via bot commands
 * 
 * Features:
 * - Smart fallback between sources
 * - Throttled updates to save battery/DB
 * - Broadcast-first architecture
 * - Background tracking via bot when Mini App closed
 */

import { useCallback, useEffect, useRef, useState } from "react";
import { useTelegram, type TelegramLocation } from "./useTelegram";
import { getSupabaseBrowserClient } from "@/lib/supabase-browser";

const THROTTLE_MS = 3000; // Minimum time between updates
const MIN_DISTANCE_M = 15; // Minimum distance to trigger update
const BATCH_INTERVAL_MS = 5000; // How often to batch-send points

export interface VibeyLocation extends TelegramLocation {
  timestamp: number;
  source: "miniapp" | "telegram_bot" | "manual";
}

export interface UseVibeyLocationOptions {
  crewSlug: string;
  userId: string;
  enabled?: boolean;
  onLocationUpdate?: (location: VibeyLocation) => void;
  onBatchSend?: (locations: VibeyLocation[]) => void;
}

export interface UseVibeyLocationReturn {
  // Current state
  currentLocation: VibeyLocation | null;
  isTracking: boolean;
  isSupported: boolean;
  accuracy: "high" | "medium" | "low" | null;
  
  // Controls
  startTracking: () => void;
  stopTracking: () => void;
  requestSingleLocation: () => Promise<VibeyLocation | null>;
  
  // Stats
  stats: {
    totalUpdates: number;
    lastUpdateAt: number | null;
    distanceTraveled: number; // meters
    avgSpeed: number; // km/h
  };
  
  // Bot integration
  requestBotLocation: () => void;
  startBotLiveLocation: () => void;
  stopBotLiveLocation: () => void;
}

// Calculate distance between two points in meters
function haversineMeters(a: { lat: number; lng: number }, b: { lat: number; lng: number }): number {
  const R = 6371e3; // Earth radius in meters
  const φ1 = (a.lat * Math.PI) / 180;
  const φ2 = (b.lat * Math.PI) / 180;
  const Δφ = ((b.lat - a.lat) * Math.PI) / 180;
  const Δλ = ((b.lng - a.lng) * Math.PI) / 180;

  const aa =
    Math.sin(Δφ / 2) * Math.sin(Δφ / 2) +
    Math.cos(φ1) * Math.cos(φ2) * Math.sin(Δλ / 2) * Math.sin(Δλ / 2);
  const c = 2 * Math.atan2(Math.sqrt(aa), Math.sqrt(1 - aa));

  return R * c;
}

export function useVibeyLocation(options: UseVibeyLocationOptions): UseVibeyLocationReturn {
  const { crewSlug, userId, enabled = false, onLocationUpdate, onBatchSend } = options;
  const { location: telegramLocation, haptic, isInTelegram } = useTelegram();
  
  const supabase = getSupabaseBrowserClient();
  
  // State
  const [currentLocation, setCurrentLocation] = useState<VibeyLocation | null>(null);
  const [isTracking, setIsTracking] = useState(false);
  const [accuracy, setAccuracy] = useState<"high" | "medium" | "low" | null>(null);
  const [stats, setStats] = useState({
    totalUpdates: 0,
    lastUpdateAt: null as number | null,
    distanceTraveled: 0,
    avgSpeed: 0,
  });

  // Refs for internal state
  const isTrackingRef = useRef(false);
  const lastLocationRef = useRef<VibeyLocation | null>(null);
  const lastSentRef = useRef<{ lat: number; lng: number; ts: number } | null>(null);
  const batchQueueRef = useRef<VibeyLocation[]>([]);
  const watchIdRef = useRef<number | null>(null);
  const batchIntervalRef = useRef<NodeJS.Timeout | null>(null);
  const speedsRef = useRef<number[]>([]);

  // Update refs when state changes
  useEffect(() => {
    isTrackingRef.current = isTracking;
  }, [isTracking]);

  // Send location to backend (throttled)
  const sendLocation = useCallback(async (location: VibeyLocation) => {
    const now = Date.now();
    const last = lastSentRef.current;

    // Throttle check
    if (last) {
      const distance = haversineMeters(
        { lat: last.lat, lng: last.lng },
        { lat: location.latitude, lng: location.longitude }
      );
      const timeDelta = now - last.ts;

      // Skip if too close and too soon
      if (distance < MIN_DISTANCE_M && timeDelta < THROTTLE_MS) {
        return;
      }
    }

    // Update last sent ref
    lastSentRef.current = {
      lat: location.latitude,
      lng: location.longitude,
      ts: now,
    };

    // 1. Broadcast to other riders (instant, no DB)
    await supabase.channel(`map:${crewSlug}`).send({
      type: "broadcast",
      event: "position",
      payload: {
        user_id: userId,
        crew_slug: crewSlug,
        lat: location.latitude,
        lng: location.longitude,
        speed_kmh: location.speed ? location.speed * 3.6 : 0,
        heading: location.heading || null,
        is_riding: true,
        source: location.source,
      },
    });

    // 2. Upsert to live_locations (throttled persistence)
    await supabase.from("live_locations").upsert(
      {
        user_id: userId,
        crew_slug: crewSlug,
        lat: location.latitude,
        lng: location.longitude,
        speed_kmh: location.speed ? location.speed * 3.6 : 0,
        heading: location.heading || null,
        is_riding: true,
        updated_at: new Date().toISOString(),
      },
      { onConflict: "user_id" }
    );

    // Haptic feedback on significant movement
    if (last) {
      const distance = haversineMeters(
        { lat: last.lat, lng: last.lng },
        { lat: location.latitude, lng: location.longitude }
      );
      if (distance > 100) {
        haptic.light();
      }
    }
  }, [crewSlug, userId, supabase, haptic]);

  // Process new location
  const processLocation = useCallback((location: VibeyLocation) => {
    setCurrentLocation(location);
    
    // Update stats
    setStats((prev) => {
      const lastLoc = lastLocationRef.current;
      let newDistance = prev.distanceTraveled;
      let newSpeeds = [...speedsRef.current];

      if (lastLoc) {
        const distance = haversineMeters(
          { lat: lastLoc.latitude, lng: lastLoc.longitude },
          { lat: location.latitude, lng: location.longitude }
        );
        newDistance += distance;

        // Calculate speed from distance and time
        const timeHours = (location.timestamp - lastLoc.timestamp) / 3600000;
        if (timeHours > 0) {
          const speedKmh = (distance / 1000) / timeHours;
          newSpeeds.push(speedKmh);
          if (newSpeeds.length > 10) newSpeeds.shift(); // Keep last 10
        }
      }

      speedsRef.current = newSpeeds;
      const avgSpeed = newSpeeds.length > 0
        ? newSpeeds.reduce((a, b) => a + b, 0) / newSpeeds.length
        : 0;

      return {
        totalUpdates: prev.totalUpdates + 1,
        lastUpdateAt: location.timestamp,
        distanceTraveled: newDistance,
        avgSpeed: Math.round(avgSpeed * 10) / 10,
      };
    });

    lastLocationRef.current = location;

    // Add to batch queue
    batchQueueRef.current.push(location);

    // Send immediately (throttled internally)
    sendLocation(location);

    // Callback
    onLocationUpdate?.(location);

    // Update accuracy indicator
    if (location.accuracy) {
      if (location.accuracy < 10) setAccuracy("high");
      else if (location.accuracy < 50) setAccuracy("medium");
      else setAccuracy("low");
    }
  }, [sendLocation, onLocationUpdate]);

  // Start tracking
  const startTracking = useCallback(() => {
    if (!telegramLocation.isSupported) {
      console.error("[useVibeyLocation] Geolocation not supported");
      return;
    }

    if (watchIdRef.current !== null) {
      navigator.geolocation.clearWatch(watchIdRef.current);
    }

    setIsTracking(true);
    haptic.medium();

    // Start watching
    watchIdRef.current = navigator.geolocation.watchPosition(
      (position) => {
        const location: VibeyLocation = {
          latitude: position.coords.latitude,
          longitude: position.coords.longitude,
          accuracy: position.coords.accuracy,
          altitude: position.coords.altitude || undefined,
          heading: position.coords.heading || undefined,
          speed: position.coords.speed || undefined,
          timestamp: position.timestamp,
          source: "miniapp",
        };
        processLocation(location);
      },
      (error) => {
        console.error("[useVibeyLocation] Watch error:", error);
        haptic.error();
      },
      {
        enableHighAccuracy: true,
        timeout: 10000,
        maximumAge: 5000,
      }
    );

    // Start batch interval
    batchIntervalRef.current = setInterval(() => {
      if (batchQueueRef.current.length > 0) {
        onBatchSend?.([...batchQueueRef.current]);
        batchQueueRef.current = [];
      }
    }, BATCH_INTERVAL_MS);
  }, [telegramLocation.isSupported, haptic, processLocation, onBatchSend]);

  // Stop tracking
  const stopTracking = useCallback(() => {
    setIsTracking(false);
    haptic.medium();

    if (watchIdRef.current !== null) {
      navigator.geolocation.clearWatch(watchIdRef.current);
      watchIdRef.current = null;
    }

    if (batchIntervalRef.current) {
      clearInterval(batchIntervalRef.current);
      batchIntervalRef.current = null;
    }

    // Send remaining batch
    if (batchQueueRef.current.length > 0) {
      onBatchSend?.([...batchQueueRef.current]);
      batchQueueRef.current = [];
    }
  }, [haptic, onBatchSend]);

  // Request single location
  const requestSingleLocation = useCallback(async (): Promise<VibeyLocation | null> => {
    const loc = await telegramLocation.request();
    if (!loc) return null;

    const vibeyLoc: VibeyLocation = {
      ...loc,
      timestamp: Date.now(),
      source: "miniapp",
    };

    processLocation(vibeyLoc);
    return vibeyLoc;
  }, [telegramLocation, processLocation]);

  // Bot integration - request location via bot
  const requestBotLocation = useCallback(() => {
    if (!isInTelegram) return;
    
    // Open bot with start parameter
    const botUsername = process.env.NEXT_PUBLIC_TELEGRAM_BOT_USERNAME || "oneBikePlsBot";
    const startParam = `location_${crewSlug}_${userId}`;
    const url = `https://t.me/${botUsername}?start=${startParam}`;
    
    window.open(url, "_blank");
    haptic.light();
  }, [isInTelegram, crewSlug, userId, haptic]);

  // Start bot live location
  const startBotLiveLocation = useCallback(() => {
    if (!isInTelegram) return;
    
    const botUsername = process.env.NEXT_PUBLIC_TELEGRAM_BOT_USERNAME || "oneBikePlsBot";
    const startParam = `live_${crewSlug}_${userId}`;
    const url = `https://t.me/${botUsername}?start=${startParam}`;
    
    window.open(url, "_blank");
    haptic.success();
  }, [isInTelegram, crewSlug, userId, haptic]);

  // Stop bot live location
  const stopBotLiveLocation = useCallback(async () => {
    // Remove from live_locations
    await supabase.from("live_locations").delete().eq("user_id", userId);
    
    // Update session
    await supabase
      .from("map_rider_sessions")
      .update({ status: "completed", sharing_enabled: false })
      .eq("user_id", userId)
      .eq("status", "active");

    haptic.success();
  }, [userId, supabase, haptic]);

  // Auto-start if enabled
  useEffect(() => {
    if (enabled && !isTracking) {
      startTracking();
    }
    return () => {
      if (isTracking) {
        stopTracking();
      }
    };
  }, [enabled]); // eslint-disable-line react-hooks/exhaustive-deps

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      if (watchIdRef.current !== null) {
        navigator.geolocation.clearWatch(watchIdRef.current);
      }
      if (batchIntervalRef.current) {
        clearInterval(batchIntervalRef.current);
      }
    };
  }, []);

  return {
    currentLocation,
    isTracking,
    isSupported: telegramLocation.isSupported,
    accuracy,
    startTracking,
    stopTracking,
    requestSingleLocation,
    stats,
    requestBotLocation,
    startBotLiveLocation,
    stopBotLiveLocation,
  };
}

// Hook for receiving bot location updates
export function useBotLocationUpdates(crewSlug: string, onUpdate: (location: VibeyLocation) => void) {
  const supabase = getSupabaseBrowserClient();

  useEffect(() => {
    // Subscribe to broadcast channel for bot updates
    const channel = supabase.channel(`map:${crewSlug}`);
    
    channel.on("broadcast", { event: "position" }, ({ payload }) => {
      if (payload.source === "telegram_bot") {
        onUpdate({
          latitude: payload.lat,
          longitude: payload.lng,
          speed: payload.speed_kmh ? payload.speed_kmh / 3.6 : undefined,
          heading: payload.heading,
          timestamp: Date.now(),
          source: "telegram_bot",
        });
      }
    }).subscribe();

    return () => {
      supabase.removeChannel(channel);
    };
  }, [crewSlug, onUpdate, supabase]);
}
